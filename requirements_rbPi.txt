# Raspberry Pi installation notes:
# 1. Make sure the system is up to date:
#    sudo apt-get update && sudo apt-get upgrade
#
# 2. Install system dependencies:
#    sudo apt-get install python3-pip python3-dev libatlas-base-dev libjasper-dev libharfbuzz0b libwebp6 libtiff5
#    sudo apt-get install libopenjp2-7 libjasper1 libharfbuzz0b libwebp6 libopenjp2-7 libtiff5 libjasper1 libharfbuzz0b libwebp6
#
# 3. Create venv:
#    python3 -m venv .venv_pi
#    source .venv_pi/bin/activate
#
# 4. Update pip and setuptools:
#    pip install --upgrade pip setuptools wheel
#
# 5. Install requirements (this can take 30-60 minutes on an RPi 3B+):
#    pip install -r requirements_rbPi.txt
#
# 6. If mediapipe fails, install the prebuilt package:
#    pip install --index-url https://www.piwheels.org/simple mediapipe
#
# 7. Alternative for memory issues: use temporary swap
#    sudo fallocate -l 2G /swapfile
#    sudo chmod 600 /swapfile
#    sudo mkswap /swapfile
